loadImage(path);

Load a png or jpeg image

Arguments

pathThe local image path

Returns

ImageThe image instance

function setup()
  createWindow(600, 600);
  img = loadImage('/path/to/mypic.png');
end

function draw()
  background(51);

  image(img, 0, 0);
end
