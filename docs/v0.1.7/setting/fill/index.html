fill

v0.1.7
fill();

Set the fill color for shapes.

fill(r, g, b);

Arguments

rThe red byte
gThe green byte
bThe blue byte
 fill(100, 255, 100);
fill(r, g, b, a);

Arguments

rThe red byte
gThe green byte
bThe blue byte
aThe alpha byte
 fill(100, 255, 100, 150);
fill(v);

Arguments

vThe grayscale value
 fill(100);
fill(color_name);

Arguments

color_nameThe color name
 fill('yellow');
fill(hex);

Arguments

hexThe hex color string
 fill('#88EE66');
